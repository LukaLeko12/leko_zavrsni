1. MINIMUM KOJI STUDENTI MORAJU IMATI JE IMPLEMENTIRANI CRUID
   Create-Read-Update-Insert-Delete
  
   Primjer koda:

// --- CREATE ---
// Datoteka: pacijent.c
void procjena_melanoma(void) {
    Pacijent p;
    unos_string(p.ime, sizeof(p.ime), "Ime: ");
    unos_string(p.prezime, sizeof(p.prezime), "Prezime: ");
    p.godine = unos_broja("godine: ");
    
    // ... izračun rizika ...
    
    spremi_u_datoteku_binarno(&p);    // CREATE u binarnu datoteku
    spremi_u_datoteku_tekstualno(&p); // CREATE u tekstualnu datoteku
    dodaj_pacijenta_u_listu(&lista_pacijenata, &p);  // INSERT u memoriju
}

// --- READ ---
// Datoteka: datoteke.c
int ucitaj_pacijente_iz_datoteke_binarno(PacijentList *lista) {
    FILE *fp = fopen(FILE_BINARNI, "rb");
    if (fp == NULL) return 1;
    
    Pacijent temp_pacijent;
    while (fread(&temp_pacijent, sizeof(Pacijent), 1, fp) == 1) {
        dodaj_pacijenta_u_listu(lista, &temp_pacijent);  // READ iz binarne datoteke
    }
    fclose(fp);
    return 1;
}

// --- UPDATE ---
// Datoteka: pacijent.c
void azuriraj_rizik_pacijenta(const char *prezime) {
    for (int i = 0; i < lista_pacijenata.count; i++) {
        if (strcmp(lista_pacijenata.pacijenti[i].prezime, prezime) == 0) {
            lista_pacijenata.pacijenti[i].rizik_postotak = unos_float("Novi rizik (%%) - ");
            printf("[+] Azuriran!\n");  // UPDATE podataka
            return;
        }
    }
}

// --- DELETE ---
// Datoteka: pacijent.c
void obrisi_pacijenta(const char *prezime) {
    for (int i = 0; i < lista_pacijenata.count; i++) {
        if (strcmp(lista_pacijenata.pacijenti[i].prezime, prezime) == 0) {
            for (int j = i; j < lista_pacijenata.count - 1; j++) {
                lista_pacijenata.pacijenti[j] = lista_pacijenata.pacijenti[j + 1];
            }
            lista_pacijenata.count--;  // DELETE pacijenta
            printf("[+] Pacijent obrisan!\n");
            return;
        }
    }
}

// Datoteka: pacijent.h
typedef struct {
    char ime[50];               // char - string
    char prezime[50];           // char - string
    int godine;                 // int - primitivni tip (cijeli broj)
    float rizik_postotak;       // float - primitivni tip (realni broj)
    char tip_pregleda[30];      // char - string
    char dijagnoza[100];        // char - string
    time_t datum_unosa;         // time_t - vremenska vrijednost

// Datoteka: ui.c
int unos_broja(const char *pitanje) {
    int broj;                   // int - primitivni tip
    printf("%s", pitanje);
    scanf("%d", &broj);
    getchar();
    return broj;
}

float unos_float(const char *pitanje) {
    float broj;                 // float - primitivni tip
    printf("%s", pitanje);
    scanf("%f", &broj);
    getchar();
    return broj;
}

9. MAKRO FUNKCIJE ILI INLINE FUNKCIJE

   Primjer koda:

// Datoteka: pacijent.h

// MAKRO 1 - Provjera memorije
#define PROVJERA_MEMORIJE(ptr) \
    do { \
        if ((ptr) == NULL) { \
            fprintf(stderr, "[!] GRESKA: Alokacija memorije neuspjesna!\n"); \
            perror("malloc"); \
            exit(EXIT_FAILURE); \
        } \
    } while(0)

// MAKRO 2 - Provjera datoteke
#define PROVJERA_DATOTEKE(fp) \
    do { \
        if ((fp) == NULL) { \
            fprintf(stderr, "[!] GRESKA: Datoteka ne moze biti otvorena!\n"); \
            perror("fopen"); \
        } \
    } while(0)

// MAKRO 3 - Provjera NULL
#define PROVJERA_NULL(ptr) \
    do { \
        if ((ptr) == NULL) return 0; \
    } while(0)

// Datoteka: lista.c - KORIŠTENJE MAKRA
int inicijalizuj_listu(PacijentList *lista) {
    lista->pacijenti = (Pacijent *)malloc(lista->capacity * sizeof(Pacijent));
    PROVJERA_MEMORIJE(lista->pacijenti);  // KORIŠTENJE MAKRA - zamjenjuje se sa sadrzajem makra
    return 1;
}

// Datoteka: datoteke.c - KORIŠTENJE MAKRA
void spremi_u_datoteku_binarno(const Pacijent *p) {
    if (p == NULL) return;
    FILE *fp = fopen(FILE_BINARNI, "ab");

// Datoteka: lista.c
int dodaj_pacijenta_u_listu(PacijentList *lista, const Pacijent *pacijent) {
    // Pokazivač na listu (*lista)
    // Pokazivač na pacijenta (*pacijent)

17. KORISTITI malloc(), calloc(), realloc(), free()
    Sve funkcije za dinamičko upravljanje memorijom

    Primjer koda:

// Datoteka: lista.c

// MALLOC - Bazična alokacija
lista->pacijenti = (Pacijent *)malloc(lista->capacity * sizeof(Pacijent));

// REALLOC - Prosljedjenje memorije kada se popuni
int prosliri_listu(PacijentList *lista) {
    int nova_kapaciteta = lista->capacity * 2;
    
    // REALLOC - Prosljedjenje postojeće memorije
    Pacijent *temp = (Pacijent *)realloc(lista->pacijenti, 
                                         nova_kapaciteta * sizeof(Pacijent));
    PROVJERA_MEMORIJE(temp);
    
    lista->pacijenti = temp;
    lista->capacity = nova_kapaciteta;

int kopiraj_datoteku(const char *izvor, const char *odrediste) {
    if (izvor == NULL || odrediste == NULL) return 0;
    
    // FOPEN - Otvori izvornu datoteku
    FILE *src = fopen(izvor, "rb");
    if (src == NULL) {
        printf("[!] Greska pri otvaranju izvorne datoteke!\n");
        perror("fopen");
        return 0;
    }
    
    // FOPEN - Otvori ciljnu datoteku
    FILE *dst = fopen(odrediste, "wb");
    if (dst == NULL) {
        printf("[!] Greska pri otvaranju odredisne datoteke!\n");
        perror("fopen");
        fclose(src);
        return 0;
    }
    
    // KOPIRAJ - Citaj i pisi u manjim dijelovima
    char buffer[BUFFER_SIZE];
    size_t bytes;
    
    while ((bytes = fread(buffer, 1, BUFFER_SIZE, src)) > 0) {
        fwrite(buffer, 1, bytes, dst);
    }
    
    if (ferror(src) || ferror(dst)) {
        printf("[!] Greska tijekom kopiranja!\n");
        perror("read/write");
        fclose(src);
        fclose(dst);
        return 0;
    }
    
    fclose(src);
    fclose(dst);
    printf("[+] Datoteka kopirana: %s -> %s\n", izvor, odrediste);
    return 1;
}

// CVOR - Sadrži podatke i pokazivače na prije i nakon
typedef struct Node {
    Pacijent data;
    struct Node *next;  // Pokazivač na sljedeći čvor
    struct Node *prev;  // Pokazivač na prethodni čvor
} Node;

// LISTA - Sadrži head, tail i brojač
typedef struct {
    Node *head;  // Pokazivač na prvi čvor
    Node *tail;  // Pokazivač na zadnji čvor
    int count;   // Broj čvorova
} DvostrukaPovezanaLista;

// Datoteka: lista.c

// DODAJ U POVEZANU LISTU
int dodaj_u_povezanu_listu(DvostrukaPovezanaLista *lista, const Pacijent *pacijent) {
    Node *novi_cvor = (Node *)malloc(sizeof(Node));
    
    novi_cvor->data = *pacijent;
    novi_cvor->next = NULL;
    novi_cvor->prev = lista->tail;  // Povezivanje sa prethodnim
    
    if (lista->tail != NULL) {
        lista->tail->next = novi_cvor;  // Povezivanje sa sljedečim
    }
    lista->tail = novi_cvor;
    
    if (lista->head == NULL) {
        lista->head = novi_cvor;
    }
    
    lista->count++;
    return 1;
}
